# 短视频平台倍速播放功能实现文档

---

## 一、功能概述

实现视频播放器的**变速不变调**功能，支持 0.5x / 1.0x / 1.5x / 2.0x 四档倍速切换：
- **音频**：使用 FFmpeg `atempo` 滤镜实现变速不变调
- **视频**：使用 FFmpeg `setpts` 滤镜调整时间戳
- **同步**：以音频时钟为主时钟，视频动态追赶

## 三、核心实现

### 3.1 setSpeed() - 倍速切换入口

```cpp
// videoplayer.cpp - setSpeed 方法
void VideoPlayer::setSpeed(double speed)
{
    // 限制速度范围
    if(speed < 0.25) speed = 0.25;
    if(speed > 4.0) speed = 4.0;
    
    // 如果速度没变，不需要重建滤镜
    if(fabs(m_videoState.playbackSpeed - speed) < 0.001) return;
    
    // 只标记请求，不立即重建（避免线程安全问题）
    m_videoState.pending_speed = speed;
    m_videoState.audio_filter_rebuild = 1;
    m_videoState.video_filter_rebuild = 1;
    qDebug() << "请求切换倍速至:" << speed << "x";
}
```

### 3.2 init_audio_filter() - 音频滤镜初始化

```cpp
// videoplayer.cpp - 音频滤镜初始化函数
bool init_audio_filter(VideoState *is) {
    // 销毁旧滤镜图
    if (is->audio_filter_graph) {
        avfilter_graph_free(&is->audio_filter_graph);
        is->audio_filter_src = nullptr;
        is->audio_filter_sink = nullptr;
    }

    is->audio_filter_graph = avfilter_graph_alloc();
    
    // 创建 abuffer 源滤镜
    const AVFilter *abuffersrc = avfilter_get_by_name("abuffer");
    char args[512];
    snprintf(args, sizeof(args),
             "time_base=1/%d:sample_rate=%d:sample_fmt=%s:channel_layout=0x%llX",
             is->pAudioCodecCtx->sample_rate,
             is->pAudioCodecCtx->sample_rate,
             av_get_sample_fmt_name(is->pAudioCodecCtx->sample_fmt),
             (long long)(is->pAudioCodecCtx->channel_layout ? 
                         is->pAudioCodecCtx->channel_layout : 
                         av_get_default_channel_layout(is->pAudioCodecCtx->channels)));

    avfilter_graph_create_filter(&is->audio_filter_src, abuffersrc, "in", args, NULL, is->audio_filter_graph);

    // 创建 abuffersink 输出滤镜
    const AVFilter *abuffersink = avfilter_get_by_name("abuffersink");
    avfilter_graph_create_filter(&is->audio_filter_sink, abuffersink, "out", NULL, NULL, is->audio_filter_graph);

    // 串联 atempo 滤镜（支持超过2.0的倍速）
    AVFilterContext *last = is->audio_filter_src;
    double remaining = is->playbackSpeed;
    
    while (remaining > 2.0) {
        AVFilterContext *atempo;
        avfilter_graph_create_filter(&atempo, avfilter_get_by_name("atempo"),
                                     "atempo1", "tempo=2.0", NULL, is->audio_filter_graph);
        avfilter_link(last, 0, atempo, 0);
        last = atempo;
        remaining /= 2.0;
    }
    
    if (fabs(remaining - 1.0) > 0.001) {
        char tmp[64];
        snprintf(tmp, sizeof(tmp), "tempo=%.4f", remaining);
        AVFilterContext *atempo;
        avfilter_graph_create_filter(&atempo, avfilter_get_by_name("atempo"),
                                     "atempo_last", tmp, NULL, is->audio_filter_graph);
        avfilter_link(last, 0, atempo, 0);
        last = atempo;
    }

    avfilter_link(last, 0, is->audio_filter_sink, 0);
    avfilter_graph_config(is->audio_filter_graph, NULL);
    
    return true;
}
```

### 3.3 init_video_filter() - 视频滤镜初始化

```cpp
// videoplayer.cpp - 视频滤镜初始化函数
bool init_video_filter(VideoState *is) {
    if (is->video_filter_graph) {
        avfilter_graph_free(&is->video_filter_graph);
        is->video_filter_src = nullptr;
        is->video_filter_sink = nullptr;
    }

    is->video_filter_graph = avfilter_graph_alloc();
    
    // 创建 buffer 源滤镜
    const AVFilter *buffersrc = avfilter_get_by_name("buffer");
    char args[512];
    snprintf(args, sizeof(args),
             "video_size=%dx%d:pix_fmt=%d:time_base=%d/%d:pixel_aspect=1/1",
             is->pCodecCtx->width, is->pCodecCtx->height,
             is->pCodecCtx->pix_fmt,
             is->pCodecCtx->time_base.num, is->pCodecCtx->time_base.den);

    avfilter_graph_create_filter(&is->video_filter_src, buffersrc, "in", args, NULL, is->video_filter_graph);

    // 创建 buffersink 输出滤镜
    const AVFilter *buffersink = avfilter_get_by_name("buffersink");
    avfilter_graph_create_filter(&is->video_filter_sink, buffersink, "out", NULL, NULL, is->video_filter_graph);

    // 创建 setpts 滤镜（调整时间戳）
    char pts_args[64];
    snprintf(pts_args, sizeof(pts_args), "setpts=%.4f*PTS", 1.0 / is->playbackSpeed);
    
    AVFilterContext *setpts;
    avfilter_graph_create_filter(&setpts, avfilter_get_by_name("setpts"),
                                 "setpts", pts_args, NULL, is->video_filter_graph);
    
    avfilter_link(is->video_filter_src, 0, setpts, 0);
    avfilter_link(setpts, 0, is->video_filter_sink, 0);
    
    avfilter_graph_config(is->video_filter_graph, NULL);
    
    return true;
}
```

### 3.4 audio_decode_frame() - 音频解码与滤镜应用

```cpp
// videoplayer.cpp - 音频解码函数关键片段
int audio_decode_frame(VideoState *is, uint8_t *audio_buf, int buf_size)
{
    // ① 处理音频滤镜重建请求（音频线程安全点）
    if (is->audio_filter_rebuild) {
        SDL_LockAudio();
        if (is->audio_filter_rebuild) {
            is->playbackSpeed = is->pending_speed;
            init_audio_filter(is);
            is->audio_filter_rebuild = 0;
        }
        SDL_UnlockAudio();
    }

    // ... 解码循环 ...
    
    // 使用 atempo 滤镜处理音频帧
    AVFrame *filtered_frame = audioFrame;
    if(is->audio_filter_graph && is->audio_filter_src && is->playbackSpeed != 1.0) {
        av_buffersrc_add_frame(is->audio_filter_src, audioFrame);

        // ② 循环取出所有输出帧（atempo可能产生多个输出或缓冲）
        AVFrame *temp_frame = av_frame_alloc();
        int got_frame = 0;
        while ((got_frame = av_buffersink_get_frame(is->audio_filter_sink, temp_frame)) >= 0) {
            // ③ 使用输出帧的 PTS 更新音频时钟（关键修复）
            if (temp_frame->pts != AV_NOPTS_VALUE) {
                is->audio_clock = temp_frame->pts * av_q2d(is->audio_st->time_base) * 1000000;
            } else if (audioFrame->pts != AV_NOPTS_VALUE) {
                is->audio_clock = audioFrame->pts * av_q2d(is->audio_st->time_base) * 1000000;
            }

            // 只保留最后一个输出帧用于重采样
            if (filtered_frame != audioFrame) {
                av_frame_unref(filtered_frame);
            }
            filtered_frame = temp_frame;
            temp_frame = av_frame_alloc();
        }
        av_frame_free(&temp_frame);

        // 如果没有任何输出帧，说明 atempo 需要更多输入
        if (filtered_frame == audioFrame) {
            return -1;
        }
    } else {
        // 未使用滤镜，直接用解码帧的 PTS 更新时钟
        if (audioFrame->pts != AV_NOPTS_VALUE) {
            is->audio_clock = audioFrame->pts * av_q2d(is->audio_st->time_base) * 1000000;
        }
    }

    // 重采样输出...
}
```

### 3.5 video_thread() - 视频线程与同步

```cpp
// videoplayer.cpp - 视频线程关键片段
int video_thread(void *arg) {
    while(1) {
        // ① 处理视频滤镜重建请求（视频线程安全点）
        if (is->video_filter_rebuild) {
            SDL_LockAudio();
            if (is->video_filter_rebuild) {
                init_video_filter(is);
                is->video_filter_rebuild = 0;
            }
            SDL_UnlockAudio();
        }

        // ... 解码循环 ...

        // 使用 setpts 滤镜处理视频帧
        AVFrame *filtered_frame = pFrame;
        if(is->video_filter_graph && is->video_filter_src && is->playbackSpeed != 1.0) {
            av_buffersrc_add_frame(is->video_filter_src, pFrame);
            AVFrame *temp_frame = av_frame_alloc();
            if(av_buffersink_get_frame(is->video_filter_sink, temp_frame) >= 0) {
                filtered_frame = temp_frame;
            } else {
                av_frame_free(&temp_frame);
                filtered_frame = pFrame;
            }
        }

        // PTS 计算（使用 filtered_frame 的 pts）
        int64_t pts = filtered_frame->best_effort_timestamp;
        double video_pts = pts * av_q2d(time_base) * 1000000;
        is->video_clock = video_pts;

        // ② 同步等待（公式修正）
        if (is->audioq != NULL) {
            int syncWaitCount = 0;
            while(1) {
                if( is->quit ) break;
                if( is->audioq->size == 0 ) break;
                double audio_pts = is->audio_clock;   // 已经是 atempo 后的时钟
                // 此时 audio_pts 就是实际播放时间，视频 pts 是 setpts 缩放后的时间，
                // 两者应该直接比较，不需要再乘 speed
                if (video_pts <= audio_pts) break;
                if( ++syncWaitCount > 400 ) break;
                SDL_Delay(5);
            }
        }

        // 渲染...
    }
}
```

## 四、修复的关键问题

| 问题 | 原因 | 修复方案 |
|------|------|----------|
| **滤镜重建线程不安全** | `setSpeed()` 直接销毁重建滤镜，音频回调/视频线程可能正在使用旧指针 | 使用标志位延迟重建，在解码线程安全点执行 |
| **音频时钟更新错误** | 使用原始解码帧的 PTS 更新时钟，而非 `atempo` 输出帧的 PTS | 使用滤镜输出帧的 PTS 更新 `audio_clock` |
| **`atempo` 输出帧没取干净** | 只取一次 `av_buffersink_get_frame`，导致缓冲累积 | 使用 `while` 循环取出所有可用帧 |
| **同步公式错误** | `v_pts <= audio_pts * playbackSpeed` 重复乘以速度因子 | 直接比较：`v_pts <= audio_pts` |

## 五、技术亮点（面试向）

1. **专业的变速方案**：使用 FFmpeg `atempo` 滤镜实现变速不变调，而非简单的采样率调整
2. **线程安全设计**：通过标志位延迟重建滤镜，避免多线程竞争
3. **atempo 串联设计**：支持超过 2.0 的倍速（通过串联多个 `atempo` 滤镜）
4. **动态滤镜重建**：播放过程中可实时切换倍速
5. **精确同步**：以处理后的音频时钟为主时钟，视频动态追赶
